comandos git

git 
    init --> inicia un repositorio git en la ruta indicada
    status --> muestra las carpetas y archivos de la ruta
    add . --> agrega los todos los archivos al branch
    add "archivo" --> agrega solo ese archivo al branch
    commit -m "etiqueta"  --> etiqueto mi repositorio de GIT 
    log --oneline --> reviso los logs de mi repositorio
    config --global user.name "mi nombre" --> configura mi nombre de manera global en mi pc
    config --global user.mail "mi email" --> configura mi email de manera global en mi pc
    config user.name "mi nombre" --> muestra el nombre que esta configurado en la pc
    config user.mail "mi email" --> muestra el email que esta configurado en la pc
    help --> muestra todos los comandos con sus explicaciones de que hace cada uno
    git clone "url brindado por github" --> recupero todo los commit que tenia en github
